On master node:
'''
sudo kubeadm init --apiserver-advertise-address=<ip of master node>

sudo useradd user -G sudo -m -s /bin/bash
$ sudo passwd user

mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config
'''
